const CONFIG = {
  FORM_URL: "[LINK_FORM]",
  SPREADSHEET_URL: "[LINK_SHEET]",
  SHEET_NAME: "Sheet1",
  SENDER_NAME: "Meta Verified Team",
  BATCH_LIMIT: 4,
  REST_MIN: 3,
  REST_MAX: 10,
  MIN_SLEEP: 30000,
  MAX_SLEEP: 60000,
  MAX_EXECUTION_TIME: 280000,
  QUOTA_RESERVE: 20,
  MAX_RETRY: 2,
  LOCK_TIMEOUT: 30000
};

function shareSingleFormToList(e) {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(CONFIG.LOCK_TIMEOUT)) {
    Logger.log("🔒 Instance khác đang chạy");
    return;
  }
  try {
    _mainLogic(e);
  } catch (err) {
    Logger.log("🔥 Lỗi: " + err.message);
  } finally {
    lock.releaseLock();
  }
}

function buildPlainText(intro, linkText, formUrl, email) {
  return (
`${intro}

${linkText} ${formUrl}

بعد إرسال الطلب، سيتم مراجعة معلوماتكم وسيتم إشعاركم بالنتيجة في أقرب وقت ممكن.

نحن نُقدّر دعمكم وتواصلكم معنا.

مع خالص التحية،
--
تم إرسال هذه الرسالة إلى ${email} لأنكم قمتم بالاشتراك في تلقي التحديثات من خدماتنا.`
  );
}

function _mainLogic(e) {
  const startTime = Date.now();
  const props = PropertiesService.getScriptProperties();
  const userEmail = getEffectiveUserEmail();
  const quota = MailApp.getRemainingDailyQuota();

  Logger.log(`📊 Quota còn: ${quota}`);
  if (quota <= CONFIG.QUOTA_RESERVE) {
    Logger.log("⚠️ Không đủ quota, dừng");
    return;
  }

  if (e === undefined) {
    props.deleteProperty("LAST_SENT_ROW");
    props.deleteProperty("SENT_EMAILS");
    deleteTriggers("shareSingleFormToList");
    Logger.log("🔄 Reset thủ công");
  }

  const ss = SpreadsheetApp.openById(extractSpreadsheetId(CONFIG.SPREADSHEET_URL));
  const sheet = ss.getSheetByName(CONFIG.SHEET_NAME);
  if (!sheet) throw new Error("Sheet không tồn tại: " + CONFIG.SHEET_NAME);

  const startRow = Number(props.getProperty("LAST_SENT_ROW")) || 2;
  const lastRow = sheet.getLastRow();

  Logger.log(`📋 Row ${startRow} → ${lastRow}`);
  if (startRow > lastRow) {
    props.deleteProperty("LAST_SENT_ROW");
    Logger.log("🎉 Gửi hết danh sách");
    return;
  }

  const rowsToProcess = Math.min(
    CONFIG.BATCH_LIMIT,
    lastRow - startRow + 1,
    quota - CONFIG.QUOTA_RESERVE
  );

  const data = sheet.getRange(startRow, 1, rowsToProcess, 7).getValues();
  const output = [];
  const sentCache = new Set(JSON.parse(props.getProperty("SENT_EMAILS") || "[]"));
  const newlySent = [];
  let processed = 0;

  for (let i = 0; i < data.length; i++) {
    if (Date.now() - startTime > CONFIG.MAX_EXECUTION_TIME) {
      Logger.log("⏱️ Gần timeout, dừng batch");
      break;
    }
    
    const row      = startRow + i;
    const email    = String(data[i][0]).trim();
    const pageName = String(data[i][1]).trim();
    let subject    = String(data[i][2]).trim();
    let intro      = String(data[i][3]).trim();
    let linkText   = String(data[i][4]).trim();
    const status   = String(data[i][5]).trim();
    const oldTime  = data[i][6];

    if (status === "✅ Sent") {
      output.push([status, oldTime]);
      processed++;
      continue;
    }

    if (!isValidEmail(email) || !pageName) {
      Logger.log(`🚫 Row ${row}: invalid — "${email}"`);
      output.push(["🚫 Invalid", new Date()]);
      processed++;
      continue;
    }

    if (sentCache.has(email)) {
      Logger.log(`⚠️ Row ${row}: duplicate — ${email}`);
      output.push(["⚠️ Duplicate", new Date()]);
      processed++;
      continue;
    }

    subject  = subject.replace(/\[Name\]/g, pageName);
    intro    = intro.replace(/\[Name\]/g, pageName).replace(/\\n/g, "\n");
    linkText = linkText.replace(/\[Name\]/g, pageName).replace(/\\n/g, "\n");

    const now = new Date();
    const dateStr = Utilities.formatDate(now, "Asia/Dubai", "dd/MM/yyyy HH:mm");
    subject = `[${dateStr}] ${subject}`;

    const body = buildPlainText(intro, linkText, CONFIG.FORM_URL, email);

    const emailOptions = {
      to: email,        
      subject: subject,
      body: body,
      name: CONFIG.SENDER_NAME,
      replyTo: userEmail  
    };

    let success = false, errorMsg = "";
    for (let attempt = 1; attempt <= CONFIG.MAX_RETRY; attempt++) {
      try {
        MailApp.sendEmail(emailOptions);
        success = true;
        Logger.log(`✅ Row ${row} | ${email}`);
        break;
      } catch (err) {
        errorMsg = err.message;
        Logger.log(`❌ Attempt ${attempt} | ${email} | ${errorMsg}`);
        Utilities.sleep(5000);
      }
    }

    if (success) {
      output.push(["✅ Sent", new Date()]);
      sentCache.add(email);
      newlySent.push(email);
    } else {
      output.push([`❌ ${errorMsg.substring(0, 50)}`, new Date()]);
    }

    if (i < data.length - 1) {
      const sleepMs = Math.floor(
        Math.random() * (CONFIG.MAX_SLEEP - CONFIG.MIN_SLEEP + 1)
      ) + CONFIG.MIN_SLEEP;
      Logger.log(`😴 Sleep ${(sleepMs / 1000).toFixed(0)}s`);
      Utilities.sleep(sleepMs);
    }

    processed++;
  }

  if (output.length > 0) {
    sheet.getRange(startRow, 6, output.length, 2).setValues(output);
    SpreadsheetApp.flush();
  }

  if (newlySent.length > 0) {
    props.setProperty("SENT_EMAILS", JSON.stringify([...sentCache]));
  }

  const nextRow = startRow + processed;
  if (nextRow <= lastRow) {
    props.setProperty("LAST_SENT_ROW", nextRow.toString());
    deleteTriggers("shareSingleFormToList");
    const restMin = Math.floor(Math.random() * (CONFIG.REST_MAX - CONFIG.REST_MIN + 1)) + CONFIG.REST_MIN;
    ScriptApp.newTrigger("shareSingleFormToList")
      .timeBased()
      .after(restMin * 60 * 1000)
      .create();
    Logger.log(`⏳ Đã đặt lịch gửi tiếp theo tại row: ${nextRow} sau ${restMin} phút`);
  } else {
    props.deleteProperty("LAST_SENT_ROW");
    props.deleteProperty("SENT_EMAILS");
    deleteTriggers("shareSingleFormToList");
    Logger.log("🎉 Hoàn thành gửi toàn bộ danh sách");
  }
}

function getEffectiveUserEmail() {
  try {
    return Session.getEffectiveUser().getEmail() || "";
  } catch (e) {
    return "";
  }
}

function extractSpreadsheetId(url) {
  const match = url.match(/\/d\/([a-zA-Z0-9_-]+)/);
  if (!match) throw new Error("URL Spreadsheet không hợp lệ");
  return match[1];
}

function isValidEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

function deleteTriggers(fn) {
  ScriptApp.getProjectTriggers().forEach(t => {
    if (t.getHandlerFunction() === fn) ScriptApp.deleteTrigger(t);
  });
}
