function shareSingleFormToList(e) {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(CONFIG.LOCK_TIMEOUT)) {
    Logger.log("🔒 Hệ thống đang bận, sẽ thử lại sau.");
    return;
  }

  try {
    _mainLogic(e);
  } catch (err) {
    Logger.log("🔥 Lỗi hệ thống: " + err.message);
  } finally {
    lock.releaseLock();
  }
}

function _mainLogic(e) {
  const startTime = Date.now();
  const props = PropertiesService.getScriptProperties();
  const userEmail = getEffectiveUserEmail();

  const quota = MailApp.getRemainingDailyQuota();
  Logger.log("📊 Quota còn lại: " + quota);
  
  if (quota <= CONFIG.QUOTA_RESERVE) {
    Logger.log("⚠️ Quota đã chạm ngưỡng an toàn. Dừng tiến trình.");
    return;
  }

  if (e === undefined) {
    props.deleteProperty("LAST_SENT_ROW");
  }

  const ss = SpreadsheetApp.openById(extractSpreadsheetId(CONFIG.SPREADSHEET_URL));
  const sheet = ss.getSheetByName(CONFIG.SHEET_NAME);

  if (!sheet) throw new Error("❌ Không tìm thấy Sheet cấu hình.");

  const startRow = Number(props.getProperty("LAST_SENT_ROW")) || 2;
  const lastRow = sheet.getLastRow();

  if (startRow > lastRow) {
    props.deleteProperty("LAST_SENT_ROW");
    Logger.log("🎉 Đã hoàn tất chiến dịch gửi email.");
    return;
  }

  const rowsToProcess = Math.min(
    CONFIG.BATCH_LIMIT,
    lastRow - startRow + 1,
    quota - CONFIG.QUOTA_RESERVE
  );

  const data = sheet.getRange(startRow, 1, rowsToProcess, 7).getValues();
  const output = [];
  const sentCache = new Set();
  let processed = 0;

  for (let i = 0; i < data.length; i++) {
    if (Date.now() - startTime > CONFIG.MAX_EXECUTION_TIME) break;
    if (MailApp.getRemainingDailyQuota() <= CONFIG.QUOTA_RESERVE) break;

    const email = String(data[i][0]).trim();
    const pageName = String(data[i][1]).trim();
    let subject = String(data[i][2]).trim();
    let intro = String(data[i][3]).trim();
    let linkText = String(data[i][4]).trim();
    const status = String(data[i][5]).trim();
    const oldTime = data[i][6];

    if (status === "✅ Sent") {
      output.push([status, oldTime]);
      processed++;
      continue;
    }

    if (!isValidEmail(email) || !pageName) {
      output.push(["🚫 Dữ liệu không hợp lệ", new Date()]);
      processed++;
      continue;
    }

    if (sentCache.has(email)) {
      output.push(["⚠️ Trùng lặp email", new Date()]);
      processed++;
      continue;
    }
    sentCache.add(email);

    subject = subject.replace(/\[Name\]/g, pageName);
    intro = intro.replace(/\[Name\]/g, pageName).replace(/\\n/g, "\n");
    linkText = linkText.replace(/\[Name\]/g, pageName);
    const randomFormUrl = getRandomFormUrl();
    const plainTextMessage = buildPlainTextEmail(
      email,
      intro,
      linkText,
      randomFormUrl
    );

    let success = false;
    let errorMsg = "";

    Logger.log(`📤 Đang gửi tới dòng ${startRow + i} | ${email}`);

    for (let attempt = 1; attempt <= CONFIG.MAX_RETRY; attempt++) {
      try {
        MailApp.sendEmail({
          to: email,
          subject: subject,
          body: plainTextMessage,
          name: CONFIG.SENDER_NAME,
          replyTo: userEmail
        });
        success = true;
        break;
      } catch (err) {
        errorMsg = err.message;
        Logger.log(`❌ Thử lần ${attempt} thất bại: ${email} | ${errorMsg}`);
        Utilities.sleep(5000);
      }
    }

    if (success) {
      output.push(["✅ Sent", new Date()]);
    } else {
      output.push(["❌ " + errorMsg, new Date()]);
    }

    const sleepTime = Math.floor(Math.random() * (CONFIG.MAX_SLEEP - CONFIG.MIN_SLEEP + 1)) + CONFIG.MIN_SLEEP;
    Utilities.sleep(sleepTime);
    processed++;
  }

  if (output.length > 0) {
    sheet.getRange(startRow, 6, output.length, 2).setValues(output);
  }

  const nextRow = startRow + processed;

  if (nextRow <= lastRow) {
    props.setProperty("LAST_SENT_ROW", nextRow.toString());
    deleteTriggers("shareSingleFormToList");

    const restMinutes = Math.floor(Math.random() * (CONFIG.REST_MAX - CONFIG.REST_MIN + 1)) + CONFIG.REST_MIN;
    ScriptApp.newTrigger("shareSingleFormToList")
      .timeBased()
      .after(restMinutes * 60 * 1000)
      .create();

    Logger.log(`⏳ Tạm nghỉ. Sẽ chạy tiếp từ dòng ${nextRow} sau ${restMinutes} phút.`);
  } else {
    props.deleteProperty("LAST_SENT_ROW");
    deleteTriggers("shareSingleFormToList");
    Logger.log("🎉 HOÀN TẤT CHIẾN DỊCH.");
  }
}

function buildPlainTextEmail(email, intro, linkText, formUrl) {
  const footer = getRandomFooter(email);

  return `${intro}

${linkText} ${formUrl}

${footer}`;
}

function getRandomFooter(email) {
  const footers = CONFIG.FOOTERS;
  const randomFooter =
    footers[Math.floor(Math.random() * footers.length)];

  return randomFooter.replace(/\{\{email\}\}/g, email);
}

function getRandomFormUrl() {
  const urls = CONFIG.FORM_URLS;
  return urls[Math.floor(Math.random() * urls.length)];
}

function getEffectiveUserEmail() {
  try {
    const email = Session.getActiveUser().getEmail();
    if (email) return email;
  } catch (e) {}
  return "ithinkofyou.teamox3@gmail.com";
}

function extractSpreadsheetId(url) {
  const match = url.match(/\/d\/([a-zA-Z0-9_-]+)/);
  if (!match) throw new Error("❌ Invalid Spreadsheet URL");
  return match[1];
}

function isValidEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

function deleteTriggers(fn) {
  ScriptApp.getProjectTriggers().forEach(t => {
    if (t.getHandlerFunction() === fn) {
      ScriptApp.deleteTrigger(t);
    }
  });
}
