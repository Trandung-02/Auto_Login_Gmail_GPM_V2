function shareSingleFormToList(e) {
  const CONFIG = {
    FORM_URL: "[LINK_FORM]",
    SPREADSHEET_URL: "[LINK_SHEET]",
    
    SHEET_NAME: "Sheet1",
    SENDER_NAME: "Meta Verified",
    PHYSICAL_ADDRESS: "1601 Willow Rd, Menlo Park, CA 94025",
    WEBSITE: "https://www.facebook.com",

    BATCH_LIMIT: 6,
    REST_MINUTES: 5,
    MIN_SLEEP: 30000,
    MAX_SLEEP: 60000,
    MAX_EXECUTION_TIME: 280000
  };

  const startTime = new Date().getTime();
  const userEmail = Session.getActiveUser().getEmail();
  var props = PropertiesService.getScriptProperties();

  if (e === null) {
    props.deleteProperty("LAST_SENT_ROW");
    deleteTriggers("shareSingleFormToList");
    Logger.log("🚀 RESET: Restarting from row 2.");
  }

  // ✅ OPEN FILE USING URL (NO NEED TO OPEN FROM ACTIVE SHEET)
  var ss;
  try {
    ss = SpreadsheetApp.openByUrl(CONFIG.SPREADSHEET_URL);
  } catch (err) {
    return Logger.log("❌ Unable to open Google Sheet: " + err.message);
  }

  var sheet = ss.getSheetByName(CONFIG.SHEET_NAME);
  if (!sheet) return Logger.log(`❌ Sheet "${CONFIG.SHEET_NAME}" not found`);

  var startRow = Number(props.getProperty("LAST_SENT_ROW")) || 2;
  var lastRow = sheet.getLastRow();

  if (startRow > lastRow) {
    props.deleteProperty("LAST_SENT_ROW");
    Logger.log("🎉 Completed!");
    return;
  }

  Logger.log(`📧 ${userEmail} | Row ${startRow}`);

  var rowsToProcess = Math.min(CONFIG.BATCH_LIMIT, lastRow - startRow + 1);
  var dataRange = sheet.getRange(startRow, 1, rowsToProcess, 7);
  var data = dataRange.getValues();

  var outputStatus = [];
  var processedCount = 0;

  for (var i = 0; i < data.length; i++) {
    if (new Date().getTime() - startTime > CONFIG.MAX_EXECUTION_TIME) {
      Logger.log("⏳ Execution time exceeded.");
      break;
    }

    var email = data[i][0]?.toString().trim();
    var pageName = data[i][1]?.toString().trim();
    var subjectInput = data[i][2]?.toString().trim();
    var messageTemplate = data[i][3]?.toString().trim();
    var lineText = data[i][4]?.toString().trim();
    var currentStatus = data[i][5];

    var rowStatus = [];

    if (currentStatus === "✅ Sent") {
      rowStatus = [currentStatus, data[i][6]];
      outputStatus.push(rowStatus);
      continue;
    }

    if (!email || !pageName || !subjectInput || !messageTemplate) {
      rowStatus = ["⚠️ Missing required data", new Date()];
      outputStatus.push(rowStatus);
      continue;
    }

    if (!subjectInput.includes("[Name]") || !messageTemplate.includes("[Name]")) {
      rowStatus = ["🚫 Missing [Name] placeholder", new Date()];
      outputStatus.push(rowStatus);
      continue;
    }

    try {
      const cleanTemplate = messageTemplate.replace(/\\n/g, "\n");
      var finalSubject = subjectInput.replace(/\[Name\]/gi, pageName);
      var messageBody = cleanTemplate.replace(/\[Name\]/gi, pageName);

      var formLine = (lineText ? lineText + "\n" : "") + CONFIG.FORM_URL;
      var randomRef = "REF-" + Math.random().toString(36).substring(2, 8).toUpperCase();

      var professionalFooter =
`🌐 ${CONFIG.WEBSITE}
📍 ${CONFIG.PHYSICAL_ADDRESS}

Reference ID: ${randomRef} | This message was sent to ${email}.`;

      var fullMessage =
`${messageBody}

${formLine}

Thank you for being part of the Meta community.

${professionalFooter}`;

      MailApp.sendEmail({
        to: email,
        subject: finalSubject,
        body: fullMessage,
        name: CONFIG.SENDER_NAME,
        replyTo: userEmail
      });

      rowStatus = ["✅ Sent", new Date()];
      Logger.log(`✅ ${email} | Ref: ${randomRef}`);

      var sleepTime = Math.floor(Math.random() * (CONFIG.MAX_SLEEP - CONFIG.MIN_SLEEP + 1)) + CONFIG.MIN_SLEEP;
      Utilities.sleep(sleepTime);

    } catch (err) {
      Logger.log(`❌ ${startRow + i}: ${err.message}`);
      rowStatus = ["❌ " + err.message.substring(0, 50), new Date()];
    }

    outputStatus.push(rowStatus);
    processedCount++;
  }

  if (outputStatus.length > 0) {
    sheet.getRange(startRow, 6, outputStatus.length, 2).setValues(outputStatus);
  }

  var nextRow = startRow + processedCount;

  if (nextRow <= lastRow) {
    props.setProperty("LAST_SENT_ROW", nextRow.toString());

    deleteTriggers("shareSingleFormToList");

    ScriptApp.newTrigger("shareSingleFormToList")
      .timeBased()
      .after(CONFIG.REST_MINUTES * 60 * 1000)
      .create();

    Logger.log(`⏳ ${processedCount} rows processed | Next: ${nextRow}`);
  } else {
    props.deleteProperty("LAST_SENT_ROW");
    Logger.log("🎉 Fully completed!");
  }
}

function deleteTriggers(functionName) {
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === functionName) {
      ScriptApp.deleteTrigger(triggers[i]);
    }
  }
}

function getFileIdFromUrl(url) {
  const match = url.match(/\/d\/([a-zA-Z0-9_-]+)/);
  return match ? match[1] : null;
}
