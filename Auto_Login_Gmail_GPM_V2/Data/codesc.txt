const CONFIG = {
  FORM_URL: "drive.google.com/file/d/1Okn5g7YEh5hanH_CS_YlPc79BXw0ZgDE/view",
  SPREADSHEET_URL: "[LINK_SHEET]",
  SHEET_NAME: "Sheet1",
  SENDER_NAME: "Meta Verified Team",
  BATCH_LIMIT: 3,  
  REST_MIN: 5,    
  REST_MAX: 10,    
  MAX_EXECUTION_TIME: 280000, 
  QUOTA_RESERVE: 15,
  MAX_RETRY: 2,
  LOCK_TIMEOUT: 30000
};

function shareSingleFormToList(e) {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(CONFIG.LOCK_TIMEOUT)) return;
  try {
    _mainLogic(e);
  } catch (err) {
    Logger.log("🔥 Lỗi: " + err.message);
  } finally {
    lock.releaseLock();
  }
}

function _mainLogic(e) {
  const startTime = Date.now();
  const props = PropertiesService.getScriptProperties();
  const userEmail = getEffectiveUserEmail();
  const quota = MailApp.getRemainingDailyQuota();

  if (quota <= CONFIG.QUOTA_RESERVE) {
    Logger.log("⚠️ Hết quota, dừng.");
    return;
  }

  if (e === undefined && !props.getProperty("LAST_SENT_ROW")) {
    props.setProperty("LAST_SENT_ROW", "2");
    deleteTriggers("shareSingleFormToList");
    Logger.log("🔄 Bắt đầu chiến dịch mới từ dòng 2");
  }

  const ss = SpreadsheetApp.openById(extractSpreadsheetId(CONFIG.SPREADSHEET_URL));
  const sheet = ss.getSheetByName(CONFIG.SHEET_NAME);
  const startRow = Number(props.getProperty("LAST_SENT_ROW")) || 2;
  const lastRow = sheet.getLastRow();

  if (startRow > lastRow) {
    props.deleteProperty("LAST_SENT_ROW");
    deleteTriggers("shareSingleFormToList");
    Logger.log("🎉 Xong! Hoàn thành gửi toàn bộ danh sách.");
    return;
  }

  const dynamicBatchSize = randomBetween(1, CONFIG.BATCH_LIMIT);
  const data = sheet.getRange(startRow, 1, Math.min(30, lastRow - startRow + 1), 7).getValues();
  const output = [];
  let sentCount = 0;

  for (let i = 0; i < data.length; i++) {
    if (Date.now() - startTime > CONFIG.MAX_EXECUTION_TIME) {
      Logger.log("⏳ Sắp hết giới hạn thời gian chạy 6 phút, tạm dừng để chuyển sang đợt sau.");
      break;
    }

    const email = String(data[i][0]).trim();
    const pageName = String(data[i][1]).trim();
    let subject = String(data[i][2]).trim();
    let intro = String(data[i][3]).trim();
    let linkText = String(data[i][4]).trim();
    const status = String(data[i][5]).trim();

    if (!email && !pageName) {
      output.push(["", ""]);
      continue;
    }

    if (status === "✅ Sent") {
      output.push([status, data[i][6]]);
      continue;
    }

    subject = `[${getRandomDateFormat(new Date())}] ${subject.replace(/\[Name\]/g, pageName)}`;
    intro = intro.replace(/\[Name\]/g, pageName).replace(/\\n/g, "\n");
    linkText = linkText.replace(/\[Name\]/g, pageName).replace(/\\n/g, "\n");

    const body = buildPlainText(intro, linkText, CONFIG.FORM_URL, email);

    let success = false;
    for (let attempt = 1; attempt <= CONFIG.MAX_RETRY; attempt++) {
      try {
        MailApp.sendEmail({to: email, subject: subject, body: body, name: CONFIG.SENDER_NAME, replyTo: userEmail});
        success = true;
        Logger.log(`✅ Sent: ${email}`);
        break;
      } catch (err) {
        Logger.log(`❌ Attempt ${attempt} Error: ${email} - ${err.message}`);
        if (attempt < CONFIG.MAX_RETRY) Utilities.sleep(5000);
      }
    }

    if (success) {
      output.push(["✅ Sent", new Date()]);
      sentCount++;
    } else {
      output.push(["❌ Error", new Date()]);
    }

    if (sentCount >= dynamicBatchSize) break;

    Utilities.sleep(getHumanLikeDelay());
  }

  if (output.length > 0) {
    sheet.getRange(startRow, 6, output.length, 2).setValues(output);
    SpreadsheetApp.flush();
  }

  const nextRow = startRow + output.length;
  props.setProperty("LAST_SENT_ROW", nextRow.toString());
  deleteTriggers("shareSingleFormToList");

  if (nextRow <= lastRow) {
    const rest = randomBetween(CONFIG.REST_MIN, CONFIG.REST_MAX);
    ScriptApp.newTrigger("shareSingleFormToList").timeBased().after(rest * 60 * 1000).create();
    Logger.log(`⏳ Đã gửi ${sentCount} mail. Nghỉ ${rest} phút, sẽ chạy tiếp từ dòng ${nextRow}`);
  } else {
    props.deleteProperty("LAST_SENT_ROW");
    Logger.log("🎉 Hoàn tất chiến dịch.");
  }
}

function randomBetween(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

function getHumanLikeDelay() {
  const r = Math.random();
  if (r < 0.65) return randomBetween(20000, 45000);
  return randomBetween(45000, 90000);
}

function buildPlainText(intro, linkText, formUrl, email) {
  return `${intro}

${linkText} ${formUrl}

نقدّر كونك جزءًا من مجتمعنا.

تم إرسال هذه الرسالة إلى ${email} نظرًا لاشتراكك في تلقي التحديثات من خدماتنا.`;
}

function getRandomDateFormat(now) {
  const formats = ["MMM dd, yyyy HH:mm", "MM/dd/yyyy hh:mm a", "yyyy-MM-dd HH:mm"];
  const format = formats[Math.floor(Math.random() * formats.length)];
  return Utilities.formatDate(now, "Asia/Qatar", format);
}

function getEffectiveUserEmail() {
  try { return Session.getEffectiveUser().getEmail() || ""; } catch (e) { return ""; }
}

function extractSpreadsheetId(url) {
  const match = url.match(/\/d\/([a-zA-Z0-9_-]+)/);
  if (!match) throw new Error("URL không hợp lệ");
  return match[1];
}

function deleteTriggers(fn) {
  ScriptApp.getProjectTriggers().forEach(t => {
    if (t.getHandlerFunction() === fn) ScriptApp.deleteTrigger(t);
  });
}
